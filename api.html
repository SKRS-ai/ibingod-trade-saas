Request Example (Python)

import requests

url = "https://api.ibingodtrade.com/v1/intensity-index"
headers = {"Authorization": "Bearer YOUR_API_KEY"}

response = requests.get(url, headers=headers)
data = response.json()

print(f"Current Trade Intensity: {data['global_index']}%")
